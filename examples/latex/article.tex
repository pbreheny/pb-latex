\begin{algorithm}
\caption{\label{Alg:gd-lasso} Group descent algorithm for the group lasso}
  \begin{algorithmic}
    \Repeat
    \For{$j = 1, 2, \ldots, J$}
      \State $\z_j = \X_j^T\r + \bb_j$
      \State $\bb_j' \gets S\left(\z_j,\lam_j\right)$
      \State $\r' \gets \r - \X_j(\bb_j'-\bb_j)$
    \EndFor
    \Until convergence
  \end{algorithmic}
\end{algorithm}
