% Footer

  \leavevmode%
  \hbox{%
    \begin{beamercolorbox}[wd=.5\paperwidth,ht=2.5ex,dp=1ex]{author in head/foot}%
      \usebeamerfont{title in head/foot}\hspace*{2ex}\footera\hfill
      \footerb\hspace*{2ex} 
    \end{beamercolorbox}%
    \begin{beamercolorbox}[wd=.5\paperwidth,ht=2.5ex,dp=1ex]{title in head/foot}%
        \usebeamerfont{title in head/foot}\hspace*{2ex}\footerc{}\hfill
        \insertframenumber{} / \inserttotalframenumber\hspace*{2ex} 
    \end{beamercolorbox}}%
    \vskip0pt%
}
